let num : number = 1;

if (num == 1) {
    console.log("Num is equal to 1!");
}

if (num > 1) {
    console.log("Num is greater than 1!");
}

if (num >= 1) {
    console.log("Num is greater or equal to 1!");
}

if (num <= 1) {
    console.log("Num is less or equal than 1!");
}

if (num != 1) {
       console.log("Num is not equal to 1!");
 
}
